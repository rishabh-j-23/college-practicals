#include <stdio.h>
/*
* Main function to count number of digits
*/
int main() {

    int num = 0, count = 0;

    printf("Number : ");
    scanf("%d", &num);

    // loop to count no. of digits
    while(num != 0){
        count++;
        num /= 10;
    }

    printf("Number of digits : %d", count);
}
